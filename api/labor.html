请求方式：POST;

    Age：'登录名称 int',
    Sex：'登录名称 int（0表示女，1表示男）',
    Birthday：'出生日期 str',
    Address：'地址 str',
    Nationality：'民族 str',
    IDCard：'身份证号码 str',

    CompanyID：'公司ID int',

    DepartureDate：'离场日期 str',
    EntryDate：'进场日期 str',
    Hardhatnum：'安全帽号 str',
    Education：'学历 str',
    CreateTime：'创建时间 str',
    ProjectID：'项目ID int',

    IssueAuth：'发证机关 str',
    Political：'政治面貌 str',
    Train：'是否培训 str（如果培训则上传图片，否则为空字符串‘’）',
    EmerCon：'紧急联系人号码 str',
    IDP：'身份证正面照 str（如果有这是图片文件，否则图片文件为空）',

    PID：'省ID int',
    CID：'市ID int',
    DID：'区县ID int',
    SVP：'开始有效期 str',
    EVP：'结束有效期 str',
    Superiors：'上级领导 int',
    IsLeader：'是否是班组长 int',
    CloseupPhoto：'近身照 str（如果有这是图片文件，否则图片文件为空）',
    Remark：'描述 str',

    Group_list：'图片分组ID list',

成功返回值：{'code': 0, 'res': 0, 'msg': '请求成功!'}

劳工查询：已完成

URL：/labor/query;

请求方式：GET;

请求参数：{
    ProjectName: 项目名字 str (没有则为空)
    CompanyName: 公司名字 str (没有则为空)
    DID: 区ID int (没有则为0)
    CID: 市ID int (没有则为0)
    PID: 省ID int (没有则为0)
    Isbad: 是否有不良记录 int (没有则为0)
    Name: 劳工名字 str (没有则为空)
    IDCard: 身份证号码 str (没有则为空)
    Sex: 是否有不良记录 int (全部则为2)
    JobType: 工种 str (全部则为空，直接传工种名称)
    Education: 学历 str (全部则为空， 直接传学历名称)
    Page: 页数 int (默认为1)
    PageSize: 没有显示数量 int (默认为10)
}

成功返回值：{'code': 0, 'res': 0, 'msg': '请求成功!', 'labor_list':[{对象1},{对象2}]}（对象里面的内容同详细信息的内容）

失败返回值：{'code': 500xx, 'res': 0, 'msg': '错误消息！'}
