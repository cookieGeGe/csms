URL：/guarantee/cguarantee/create;

请求方式：GET;

请求参数：{
    GID: 保函ID int (新建时为0或者无此选项)
    Name: 反担保人名字 str
    IDCard: 反担保人身份证号码 str
    Address: 反担保人地址 str
    Phone: 反担保人电话 str
    Area: 房产面积 str
    Pvalue: 房产价值 str
    Proportion: 产权比例 str
    Paddress: 房产证地址 str
    IDimg: 身份证图片 filelist
    Pimg: 房产证图片 filelist
}

成功返回值：{'code': 0, 'res': 0, 'msg': '请求成功!', cguarantee:id
}
